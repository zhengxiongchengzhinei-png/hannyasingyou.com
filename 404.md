const express = require('express');
const { createProxyMiddleware } = require('http-proxy-middleware');
const app = express();

// ポート番号の設定（環境変数がない場合は3000を使用）
const PORT = process.env.PORT || 3000;

// 簡易的なフロントページ（般若心経プロキシのトップ画面）
app.get('/', (req, res) => {
    res.send(`
        <!DOCTYPE html>
        <html lang="ja">
        <head>
            <meta charset="UTF-8">
            <title>般若心経 Proxy (Hannya Shingyo)</title>
            <style>
                body { background-color: #1a1a1a; color: #d4af37; font-family: serif; text-align: center; padding-top: 50px; }
                input { width: 60%; padding: 10px; font-size: 16px; border: 1px solid #d4af37; background: #333; color: #fff; }
                button { padding: 10px 20px; font-size: 16px; background: #d4af37; color: #1a1a1a; border: none; cursor: pointer; }
            </style>
        </head>
        <body>
            <h1>摩訶般若波羅蜜多心経 Proxy</h1>
            <p>閲覧したいURLを入力してください（例: https://example.com）</p>
            <form action="/go" method="GET">
                <input type="text" name="url" placeholder="https://..." required>
                <button type="submit">度一切苦厄（接続）</button>
            </form>
        </body>
        </html>
    `);
});

// リクエストを中継するルーティング
app.get('/go', (req, res, next) => {
    const targetUrl = req.query.url;
   
    if (!targetUrl) {
        return res.status(400).send('URLが指定されていません。');
    }

    // 入力されたURLに対して動的にプロキシを適用
    createProxyMiddleware({
        target: targetUrl,
        changeOrigin: true,
        followRedirects: true,
        pathRewrite: (path, req) => {
            return ''; // ターゲットURLのルートに書き換え
        },
        onError: (err, req, res) => {
            res.status(500).send('般若心経 Proxy エラー: 目的地に到達できませんでした。');
        }
    })(req, res, next);
});
